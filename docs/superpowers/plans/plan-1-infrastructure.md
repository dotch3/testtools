> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Set up the complete project skeleton — monorepo, Docker Compose, Prisma schema (22 tables), Fastify backend boilerplate, Next.js 16.2 frontend boilerplate, storage/mail interfaces, seed data, and health check.

**Architecture:** Monorepo with `backend/` and `frontend/` packages. Backend is Fastify + TypeScript + Prisma. Frontend is Next.js 16.2 + TypeScript + Tailwind. Docker Compose with `local-db` profile for optional local PostgreSQL.

**Tech Stack:** Node.js 22, TypeScript 5, Fastify 5, Prisma 6, PostgreSQL 16, Redis 7, Next.js 16.2, Tailwind CSS, Vitest, Zod, Winston, next-themes, next-intl

# docker-compose.yml
#
# Profiles:
#   local-db  — starts postgres and backup containers (on-premise / full-local mode)
#
# Usage:
#   Full-local:  docker-compose --profile local-db up
#   Cloud DB:    docker-compose up           (postgres + backup containers are skipped)

services:
  # ── PostgreSQL (local-db profile only) ────────────────────────────────────
  postgres:
    image: postgres:16-alpine
    profiles: [local-db]
    environment:
      POSTGRES_USER: postgres
      POSTGRES_PASSWORD: postgres
      POSTGRES_DB: teststool
    volumes:
      - postgres_data:/var/lib/postgresql/data
    networks:
      - internal
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U postgres"]
      interval: 10s
      timeout: 5s
      retries: 5

  # ── Redis ─────────────────────────────────────────────────────────────────
  # Always started. Point REDIS_URL to Upstash or Redis Cloud to skip this.
  redis:
    image: redis:7-alpine
    command: redis-server --appendonly yes
    volumes:
      - redis_data:/data
    networks:
      - internal
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

  # ── Backend ───────────────────────────────────────────────────────────────
  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    env_file: .env
    ports:
      - "3001:3001"
    volumes:
      - uploads_data:/data/uploads
      - logs_data:/logs
    networks:
      - internal
      - public
    depends_on:
      redis:
        condition: service_healthy
      # postgres is only present under --profile local-db.
      # Docker Compose silently ignores this dependency when the profile is inactive.
      postgres:
        condition: service_healthy

  # ── Worker ────────────────────────────────────────────────────────────────
  # Runs the same Docker image as backend but in worker mode.
  worker:
    build:
      context: ./backend
      dockerfile: Dockerfile
    command: ["node", "dist/worker.js"]
    env_file: .env
    volumes:
      - uploads_data:/data/uploads
      - logs_data:/logs
    networks:
      - internal
    depends_on:
      redis:
        condition: service_healthy
      # Same pattern as backend — safe to declare even when postgres is not profiled in.
      postgres:
        condition: service_healthy

  # ── Frontend ──────────────────────────────────────────────────────────────
  frontend:
    build:
      context: ./frontend
      dockerfile: Dockerfile
    env_file: .env
    ports:
      - "3000:3000"
    networks:
      - public
    depends_on:
      - backend

  # ── Backup (local-db profile only) ────────────────────────────────────────
  # Runs pg_dump daily using busybox crond (installed at startup).
  # Cloud databases (Supabase, Neon, RDS) manage their own backups.
  backup:
    image: postgres:16-alpine
    profiles: [local-db]
    env_file: .env
    volumes:
      - backups_data:/data/backups
    networks:
      - internal
    entrypoint: >
      sh -c "apk add --no-cache busybox && echo '0 2 * * * pg_dump $$DATABASE_URL -F c -f /data/backups/teststool_$$(date +\\%Y\\%m\\%d_\\%H\\%M\\%S).dump' | crontab - && crond -f -l 8"
    depends_on:
      postgres:
        condition: service_healthy

volumes:
  postgres_data:
  redis_data:
  uploads_data:
  backups_data:
  logs_data:

networks:
  internal:
  public:

export type Config = z.infer<typeof configSchema>

export function validateConfig(env: NodeJS.ProcessEnv): Config {

  if (!result.success) {

// backend/src/logger.ts
// Structured JSON logger using Winston.
// Log level and output controlled by LOG_LEVEL and LOG_OUTPUT env vars.
import winston from 'winston'
import DailyRotateFile from 'winston-daily-rotate-file'
import { config } from './config.js'

const jsonFormat = winston.format.combine(
  winston.format.timestamp(),
  winston.format.errors({ stack: true }),
  winston.format.json(),
)

const transports: winston.transport[] = []

// Always add stdout if LOG_OUTPUT is stdout or both
if (config.LOG_OUTPUT === 'stdout' || config.LOG_OUTPUT === 'both') {
  transports.push(new winston.transports.Console({ format: jsonFormat }))
}

// Add rotating file transport if LOG_OUTPUT is file or both
if (config.LOG_OUTPUT === 'file' || config.LOG_OUTPUT === 'both') {
  transports.push(
    new DailyRotateFile({
      dirname: config.LOG_PATH,
      filename: 'app-%DATE%.log',
      datePattern: 'YYYY-MM-DD',
      maxSize: config.LOG_MAX_SIZE,
      maxFiles: config.LOG_MAX_FILES,
      format: jsonFormat,
    }),
  )
}

export const logger = winston.createLogger({
  level: config.LOG_LEVEL,
  transports,
})
```

// backend/prisma/schema.prisma
// Database schema for teststool — 22 tables.
// All enum-like fields use FK → EnumValue instead of Prisma enums,
// so admins can customize values from the UI.

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider          = "postgresql"
  // url = pooled connection (PgBouncer/Neon pooler) — used by the app at runtime.
  // Falls back to DATABASE_URL when DATABASE_POOL_URL is not set (full-local mode).
  url               = env("DATABASE_POOL_URL")

  directUrl         = env("DATABASE_URL")
  shadowDatabaseUrl = env("SHADOW_DATABASE_URL")
}

// ─── Auth & Users ──────────────────────────────────────────────────────────

model User {
  id                  String    @id @default(uuid())
  email               String    @unique
  name                String?
  avatarUrl           String?
  roleId              String
  role                Role      @relation(fields: [roleId], references: [id])
  passwordHash        String?
  emailVerified       Boolean   @default(false)
  lastLoginAt         DateTime?
  failedLoginCount    Int       @default(0)
  lockedUntil         DateTime?
  themePreference     String?   // dark | light | system
  forcePasswordChange Boolean   @default(false)
  createdAt           DateTime  @default(now())

// backend/src/infrastructure/storage/SupabaseStorageAdapter.ts
// Supabase Storage adapter — used when STORAGE_PROVIDER=supabase.
import { createClient } from '@supabase/supabase-js'
import type { IFileStorageAdapter } from '../../domain/interfaces/IFileStorageAdapter.js'
import type { Config } from '../../config.js'

export class SupabaseStorageAdapter implements IFileStorageAdapter {
  private readonly bucket: string
  private readonly client: ReturnType<typeof createClient>

  constructor(cfg: Config) {
    this.bucket = cfg.SUPABASE_STORAGE_BUCKET
    this.client = createClient(cfg.SUPABASE_URL!, cfg.SUPABASE_SERVICE_KEY!)
  }

  async save(path: string, buffer: Buffer, mimeType: string): Promise<string> {
    const { error } = await this.client.storage
      .from(this.bucket)
      .upload(path, buffer, { contentType: mimeType, upsert: true })
    if (error) throw new Error(`Supabase upload failed: ${error.message}`)
    return path
  }

  async delete(path: string): Promise<void> {
    const { error } = await this.client.storage.from(this.bucket).remove([path])
    if (error) throw new Error(`Supabase delete failed: ${error.message}`)
  }

  async getUrl(path: string): Promise<string> {
    const { data } = await this.client.storage
      .from(this.bucket)
      .createSignedUrl(path, 3600) // 1-hour signed URL
    if (!data?.signedUrl) throw new Error('Failed to generate signed URL')
    return data.signedUrl
  }
}

// backend/src/infrastructure/storage/S3StorageAdapter.ts
// S3-compatible storage adapter — used when STORAGE_PROVIDER=s3.
// Works with AWS S3, Cloudflare R2, and MinIO.
import {
  S3Client,
  PutObjectCommand,
  DeleteObjectCommand,
  GetObjectCommand,
} from '@aws-sdk/client-s3'
import { getSignedUrl } from '@aws-sdk/s3-request-presigner'
import type { IFileStorageAdapter } from '../../domain/interfaces/IFileStorageAdapter.js'
import type { Config } from '../../config.js'

export class S3StorageAdapter implements IFileStorageAdapter {
  private readonly client: S3Client
  private readonly bucket: string

  constructor(cfg: Config) {
    this.bucket = cfg.S3_BUCKET
    this.client = new S3Client({
      region: cfg.S3_REGION,
      // S3_ENDPOINT is required for R2/MinIO; leave empty for AWS S3
      ...(cfg.S3_ENDPOINT ? { endpoint: cfg.S3_ENDPOINT } : {}),
      credentials: {
        accessKeyId: cfg.S3_ACCESS_KEY_ID!,
        secretAccessKey: cfg.S3_SECRET_ACCESS_KEY!,
      },
    })
  }

  async save(path: string, buffer: Buffer, mimeType: string): Promise<string> {
    await this.client.send(
      new PutObjectCommand({ Bucket: this.bucket, Key: path, Body: buffer, ContentType: mimeType }),
    )
    return path
  }

  async delete(path: string): Promise<void> {
    await this.client.send(new DeleteObjectCommand({ Bucket: this.bucket, Key: path }))
  }

  async getUrl(path: string): Promise<string> {
    return getSignedUrl(
      this.client,
      new GetObjectCommand({ Bucket: this.bucket, Key: path }),
      { expiresIn: 3600 }, // 1-hour signed URL
    )
  }
}

export function createStorageAdapter(): IFileStorageAdapter {
  switch (config.STORAGE_PROVIDER) {
    case 'supabase':

        throw new Error('SUPABASE_URL and SUPABASE_SERVICE_KEY are required when STORAGE_PROVIDER=supabase')

      }
      return new S3StorageAdapter(config)

    case 'local':
    default:
      return new LocalStorageAdapter(config.STORAGE_PATH)
  }
}

# Multi-stage build using Next.js standalone output (~120MB).

# ── Stage 1: builder ──────────────────────────────────────────────────────
FROM node:22-alpine AS builder

WORKDIR /app

COPY package*.json ./
RUN npm ci

COPY . .

# Build Next.js app with standalone output
ENV NEXT_TELEMETRY_DISABLED=1
RUN npm run build

# ── Stage 2: runner ───────────────────────────────────────────────────────
FROM node:22-alpine AS runner

WORKDIR /app

ENV NODE_ENV=production
ENV NEXT_TELEMETRY_DISABLED=1

# Copy only the standalone output and static assets
COPY --from=builder /app/.next/standalone ./
COPY --from=builder /app/.next/static ./.next/static
COPY --from=builder /app/public ./public

EXPOSE 3000

CMD ["node", "server.js"]
```
